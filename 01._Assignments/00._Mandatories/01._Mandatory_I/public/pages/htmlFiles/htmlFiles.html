Nodemon

Nodemon is a tool that helps develop Node.js applications by automatically restarting the server when changes are detected in the codebase. This can save time and effort during development by eliminating the need to manually stop and restart the server after each code change.

Nodemon should only be used in development. 


      //globally
      npm install nodemon -g 
      //locally
      npm install nodemon --save-dev

To start a Node.js application with Nodemon, use the following command: 


        nodemon app.js

Anatomy of an URL


      // Anatomy of a URL
      // Example URL: https://www.example.com:80/path/to/resource?query=search#fragment
      
      // Protocol: https
      // Hostname: www.example.com
      // Port: 80
      // Path: /path/to/resource
      // Query string: ?query=search
      // Fragment: #fragment
      
      // In a Node.js context, you can use the URL module to parse URLs:
      const url = require('url');
      const myUrl = new URL('https://www.example.com:80/path/to/resource?query=search#fragment');
      
      console.log(myUrl.hostname); // Outputs: www.example.com

POST request


      // Handling a POST request with Express

      const express = require('express');
      const bodyParser = require('body-parser');
      
      const app = express();
      
      // Middleware to parse request bodies
      app.use(bodyParser.urlencoded({ extended: true }));
      app.use(bodyParser.json());
      
      app.post('/submit-data', (req, res) => {
        // Access POST data with req.body
        console.log(req.body);
        res.send('POST request to the homepage');
      });
      
      app.listen(3000, () => console.log('Server started on port 3000'));

Body parsing


      // Example of body parsing in Express.js

      const express = require('express');
      const app = express();
      
      // Built-in middleware to handle urlencoded form data
      app.use(express.urlencoded({ extended: true }));
      
      // Built-in middleware to handle JSON data
      app.use(express.json());
      
      app.post('/api/data', (req, res) => {
        // Accessing parsed body data
        console.log(req.body);
        res.json({ message: 'Data received successfully', receivedData: req.body });
      });
      
      app.listen(3000, () => {
        console.log('Server running on http://localhost:3000');
      });
