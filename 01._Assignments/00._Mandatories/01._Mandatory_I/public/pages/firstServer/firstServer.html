      }

      console.log(greeting); // Output: 'Hello, Alice!'

Callbacks


      // Example of a callback function in JavaScript
      function processData(callback) {
        // Simulate data fetching
        const data = 'fetched data';
        callback(data);
      }
      
      processData(function(data) {
        console.log('Data received:', data);
      });

Modules


      // Creating a module (myModule.js)
      module.exports = {
        greet: function(name) {
          console.log('Hello, ' + name + '!');
        }
      };
      
      // Importing a module (app.js)
      const myModule = require('./myModule');
      myModule.greet('Alice');

Express


      // Setting up a simple Express server
      const express = require('express');
      const app = express();
      
      app.get('/', (req, res) => {
        res.send('Hello World with Express!');
      });
      
      app.listen(3000, () => {
        console.log('Server is running on http://localhost:3000');
      });

GET requests


      // Handling GET requests with Express
      app.get('/api/data', (req, res) => {
        res.json({message: 'This is a response to a GET request'});
